<script>
    // State management for Chardham helicopter rates
    let chardhamState = {
        passengers: 2,
        sharingPlan: 'double', // double, triple, quad
        rates: {
            double: 261999,
            triple: 260999,
            quad: 259999
        },
        addons: {
            vipPooja: { selected: false, cost: 11000 },
            deluxeSuite: { selected: false, cost: 24999 }
        },
        tokenCostPerPax: 1999
    };

    // Sharing Plan Handler
    function setSharingPlan(plan) {
        chardhamState.sharingPlan = plan;
        
        // Visual button states update
        document.querySelectorAll('.sharing-btn').forEach(btn => {
            btn.className = "sharing-btn border border-stone-200 bg-white font-medium text-xs py-1.5 rounded-lg text-stone-600 uppercase text-center transition";
        });
        
        let activeBtn = document.getElementById(`btn-${plan}`);
        if(activeBtn) {
            activeBtn.className = "sharing-btn border-2 border-luxury-gold bg-luxury-lightGold font-bold text-xs py-1.5 rounded-lg text-luxury-dark uppercase text-center transition";
        }
        
        recalculateChardhamPrice();
    }

    // Passengers Counter Handler
    function updatePassengers(amount) {
        chardhamState.passengers = Math.max(1, chardhamState.passengers + amount);
        document.getElementById('pax-count').textContent = chardhamState.passengers;
        recalculateChardhamPrice();
    }

    // Recalculator Function
    function recalculateChardhamPrice() {
        // Fetch checkboxes
        chardhamState.addons.vipPooja.selected = document.getElementById('addon-vip-pooja').checked;
        chardhamState.addons.deluxeSuite.selected = document.getElementById('addon-deluxe-suite').checked;

        // Base flight calculation
        let baseRate = chardhamState.rates[chardhamState.sharingPlan];
        let baseTotal = baseRate * chardhamState.passengers;

        // Addon values calculation
        let addonsTotal = 0;
        if (chardhamState.addons.vipPooja.selected) {
            addonsTotal += chardhamState.addons.vipPooja.cost * chardhamState.passengers;
        }
        if (chardhamState.addons.deluxeSuite.selected) {
            addonsTotal += chardhamState.addons.deluxeSuite.cost * chardhamState.passengers;
        }

        // Final variables
        let grandTotal = baseTotal + addonsTotal;
        let tokenTotal = chardhamState.passengers * chardhamState.tokenCostPerPax;

        // Update display text values
        document.getElementById('calculated-grand-total').textContent = '₹' + grandTotal.toLocaleString('en-IN');
        document.getElementById('calculated-token-advance').textContent = '₹' + tokenTotal.toLocaleString('en-IN');
        document.getElementById('pax-sharing-label').textContent = `${chardhamState.sharingPlan} Sharing Rate Plan`;
    }

    // Scrolling utility for interactive link
    function scrollToCalculator() {
        document.getElementById('pricing-tool-section').scrollIntoView({ behavior: 'smooth' });
    }

    // Standard load execution
    window.onload = function() {
        recalculateChardhamPrice();
    };
</script>
